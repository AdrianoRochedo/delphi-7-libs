            using MySQLDriverCS;
            ...
            MySQLConnection DBConn;
            DBConn = new MySQLConnection( new MySQLConnectionString("myhost","mydatabase","mylogin","mypassword").AsString );

            ...
            new MySQLInsertCommand(
            	DBConn,

            		{"SettingValue",Value}
            	},
            	"Settings"

            ...

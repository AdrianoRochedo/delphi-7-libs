Exemplo 3: Uso de Objetos 

Program Teste;
 
Var
   i, ii  : Integer;
   S      : String;
   OutPut : Object;
   SL     : Object;
   
 begin
  OutPut := CreateObject('TOutPut');
  SL     := CreateObject('TStringList');

  OutPut.Write('Objeto OutPut');
  OutPut.Write('');

  i := 1;
  while i <= 20 do
    begin
    s := '';
    ii := 1;
    while ii <= i do
      begin
      s := s + '-';
      ii := ii + 1;
      end;
    OutPut.Write(s);
    i := i + 1;
    end;

  OutPut.Write('');
  SL.LoadFromFile('F:\Projetos\Outros\Compilador\Pasqual.cfg');
  for i := 0 to SL.Count - 1 do
    OutPut.Write(SL.GetString(i));

  OutPut.Show; 

  FreeObject(SL);
  FreeObject(OutPut);
end.
Exemplo 4: Uso de Objetos 2 

Program Teste;
 
Var
   i, ii  : Integer;
   S      : String;
   OutPut : Object;
   M, M2  : Object;
   EC     : Integer;  {ErrorCode}
   L, C   : Integer;
   
 begin
  L := 5;
  C := 5;

  OutPut := CreateObject('TOutPut');
  M      := MatrixCreate(L, C);

  {Exemplo de como acessar os elementos de uma matriz}
  for i := 1 to L do
    for ii := 1 to C do
      M.SetData(i, ii, i * ii);

  OutPut.Write('Antes da Potenciação');
  OutPut.Write('');

  M.Print(OutPut);
  M2 := M.Power(2, EC); 
  OutPut.Write('');
  
  if EC = 0 then
     begin
     OutPut.Write('Após a Potenciação por 2');
     OutPut.Write('');
     M.Print(OutPut);
     FreeObject(M2);
     end
  else
     OutPut.Write('Houve um erro na exponenciação. Código: ' + ValToStr(EC));

  OutPut.Show; 

  FreeObject(M);
  FreeObject(OutPut);
end.
Exemplo 5: Comando FOR melhorado 

Program TesteFOR;
 
Var i, ii  : Real;
    Saida  : Object;

begin
  Saida := CreateObject('TOutPut');  

  for i := 1.1 to 2.5 step 0.1 do 
    Saida.Write(ValToStr(i));

  Saida.Write('');

  for i := 2.5 downto 1.1 step 0.1 do 
    Saida.Write(ValToStr(i));

  Saida.Show;
  FreeObject(Saida);
end.
